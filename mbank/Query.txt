CREATE TABLE USER_ACC
(
    NOREK     CHAR(9)     NOT NULL,
    USERNAME  VARCHAR(20) NOT NULL,
    PASSWORD  CHAR(28)    NOT NULL,
    SALDO     INT         NOT NULL,
    EMAIL     VARCHAR(50),
    HANDPHONE VARCHAR(10),
    PRIMARY KEY (NOREK)
)

CREATE TABLE TRANSACTION_HIST
(
    ID     CHAR(5)  NOT NULL,
    REK1   CHAR(9)  NOT NULL,
    REK2   CHAR(9)  NOT NULL,
    DATE   DATETIME NOT NULL,
    AMOUNT INT      NOT NULL,
    TYPE   CHAR(3)  NOT NULL,
    PRIMARY KEY (ID),
    FOREIGN KEY (REK1) REFERENCES USER_ACC (NOREK),
    FOREIGN KEY (REK2) REFERENCES USER_ACC (NOREK)
)

CREATE TABLE CARD
(
    ID         CHAR(16)    NOT NULL,
    NOREK      CHAR(9)     NOT NULL,
    TYPE       VARCHAR(20) NOT NULL,
    CVV        CHAR(3)     NOT NULL,
    EXPIRATION CHAR(4)     NOT NULL,
    IS_PRIMARY BIT         NOT NULL,
    IS_BLOCKED BIT         NOT NULL,
    PRIMARY KEY (ID),
    FOREIGN KEY (NOREK) REFERENCES USER_ACC (NOREK)
)
